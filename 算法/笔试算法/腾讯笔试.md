```java
import java.util.*;


public class Solution {
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param heights int整型一维数组 
     * @return int整型一维数组
     */
    public int[] findBuilding (int[] heights) {
        int n = heights.length;
        int[] ans = new int[n];
        Arrays.fill(ans,1);
        LinkedList<Integer> list = new LinkedList<>();
        for(int i =0;i<n-1;i++){
           while(list.size()!=0&&heights[list.peekLast()]<=heights[i]){
               list.pollLast();
           } 
            list.addLast(i);
            ans[i+1]+=list.size();
        }
        list.clear();
        for(int i =n-1;i>0;i--){
           while(list.size()!=0&&heights[list.peekLast()]<=heights[i]){
               list.pollLast();
           } 
            list.addLast(i);
            ans[i-1]+=list.size();
        }
        return ans;
    }
}
```

```java
import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        sc.nextLine();
        int[] work = new int[n];
        int[] voc = new int[n];
        for(int i = 0;i<n;i++){
            work[i] = sc.nextInt();
        }
        sc.nextLine();
        for(int i = 0;i<n;i++){
            voc[i] = sc.nextInt();
        }
        sc.nextLine();
        int[][] dp = new int[n+1][3];//在第i天工作还是休息还是健身
        dp[0][0] = 0;        // 0休息1工作2健身
        dp[0][1] = 0;
        dp[0][2] = 0;
        for(int i = 1; i <= n; i++)
            for(int j = 0; j < 3; j++) dp[i][j] = Integer.MAX_VALUE;
        for(int i =1;i<=n;i++){
            if(work[i-1]==1){
                dp[i][1] = Math.min(dp[i-1][0],dp[i-1][2]);
            }
            if(voc[i-1]==1){
                dp[i][2] = Math.min(dp[i-1][0],dp[i-1][1]);
            }

            dp[i][0] = Math.min(dp[i-1][0],Math.min(dp[i-1][2],dp[i-1][1]))+1;
        }
        int min = Math.min(dp[n][0],Math.min(dp[n][1],dp[n][2]));
        System.out.println( min);
    }
}
```

##### 5.[编程题视野争夺](https://www.nowcoder.com/question/next?pid=21283868&qid=830863&tid=54094446)

```java

import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int L = sc.nextInt();
        sc.nextLine();
        int[][] intervals = new int[n][2];
        for(int i = 0;i<n;i++){
            intervals[i][0] = sc.nextInt();
            intervals[i][1] = sc.nextInt();
            sc.nextLine();
        }
        Arrays.sort(intervals,(a,b)->{
            if(a[0]==b[0]){
                return b[1]-a[1];
            }
            return a[0]-b[0];
        });
        int count = 0;
        int pre = 0;
        int index = 0;
        while(pre<L){
            if(intervals[index][0]>pre){
                System.out.println(-1);
            }
            int max = 0;
            while(index<intervals.length&&intervals[index][0]<=pre){
                max = Math.max(max,intervals[index][1]);
                index++;
            }
            count++;
            pre = max;
            if(pre>=L){
                System.out.println(count);
                return;
            }
            if(index>=n){
                System.out.println(-1);
                return;
            }
        }
    }
}
```
